                        import { useState } from 'react';
                        function About() {
                          const [message, setMessage] = useState('Najaf Ali');
                          function Click() {
                            setMessage('Ali Abbas!');
                          }
                          return (
                            

                              <h1>this is an About page</h1>
                                <p>{message}</p>

                              <button onClick={Click}>update me</button>
                            

                          );
                        }
                        export default About;
is code mai hum ny useState ko import kiya from react, ye isy leye kiya ta k hum data ko browser page mai show krwa sakay, useStae ye hamaye facility deta hy . jb hum useState ko use krtay hai tub hum data banatay hai jo by default hota hy phr usko set krtay hai k click pe kiya ho mean dynamic data changing krtay hai, message dafault ho gaya aur setMessage after click show ho ga message ki value ko kaha pe show krna hy us k leye hum message variable ko html mai likhtay hai jaesay uper walay code mai um ny button k upper show krwana chahtay hai message value ko typescript k variable ko html tag k andr use krnay ka tarika ye hy curly bracket k andr hum variable name likh detay hai, like <p>{message}</p>
